def KarenAndCoffee():    n, k , q = map(int, input().split())     recipes = []    for i in range(n):        l, r = map(int, input().split())        recipes.append([l,r])         questions = []    for i in range(q):        l, r = map(int, input().split())        questions.append([l,r])      min_l = float('inf')    max_r = 0    for l, r in questions:        min_l = min(l,min_l)        max_r = max(r,max_r)     freq = [0] * (max_r - min_l + 1)     for l, r in recipes:        if max_r < l or min_l > r:            continue        if l < min_l:            freq[0] += 1        else:            freq[l - min_l] += 1                    if r < max_r:            freq[(r + 1) - min_l] -= 1        for i in range(1,len(freq)):        freq[i] += freq[i-1]              for i in range(len(freq)):        if freq[i] >= k:            freq[i] = 1        else:            freq[i] = 0     for i in range(1,len(freq)):        freq[i] += freq[i-1]        num = 0    for l, r in questions:        num =  freq[r - min_l] - freq[l - min_l - 1] if l - min_l - 1 > -1 else freq[r - min_l]        print(num)     KarenAndCoffee()
